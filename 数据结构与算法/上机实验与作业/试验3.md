## 实验4.1 串的表示, 实现和基本操作

### 考核内容

+   采用定长顺序储存表示
+   实现串的基本操作
    +   求串长`StrLength()`
    +   插入运算`StrInsert()`
    +   子串删除`StrDelete()`
    +   串连接`StrConcat()`
    +   取子串`SubString()`

### 程序实现

```C
//
// Created by 张子栋 on 2021/10/15.
//

#include "stdio.h"
#include "stdlib.h"
#include "string.h"
#include "malloc.h"

#define  MAXSIZE 255

typedef char strs[MAXSIZE];
typedef struct {
    strs str;
    int length;
} SeqString;

void strInsert(SeqString S, int pos, SeqString T) {
    int k;
    if (pos >= S.length || S.length + T.length > MAXSIZE) {
        printf("不能插入\n");
    } else {
        for (k = S.length - 1; k >= pos - 1; k--) {
            S.str[T.length + k] = S.str[k];
        }
        for (k = 0; k < T.length; k++) {
            S.str[pos + k - 1] = T.str[k];
        }
        S.length = S.length + T.length;
        S.str[S.length] = "\0";
    }
    printf("插入后串长是:%d\n", S.length);
    printf("插入后的串是:");
    printf("s%\n", S.str);
    return;
}

int strLength(SeqString S) {
    int i = 0;
    while (S.str[i] != '\0') {
        i++;
    }
    S.length = i;
    printf("串长是:%d", S.length);
    return S.length;
}

SeqString *strConConcat(SeqString S, SeqString T) {
    int i;
    SeqString *r;
    if (S.length + T.length > MAXSIZE) {
        printf("不能连接!");
        return (NULL);
    } else {
        r = (SeqString *) malloc(sizeof(SeqString));
        //S复制到r中字符串数组的前端
        S.length = strlen(S.str);
        for (i = 0; i < S.length; i++) {
            r->str[i] = S.str[i];
        }
        //将T复制到r中字符串数组的后端
        T.length = strlen(S.str);
        for (i = 0; i < T.length; i++) {
            r->str[S.length + i] = T.str[i];
        }
        r->length = S.length + T.length;
        printf("连接后的串长是: ");
        printf("%d\n", r->length);
        r->str[r->length] = '\0';
    }
    printf("连接后的串是:");
    printf("%s\n", r->str);
    return (r);
}

void strDelete(SeqString *S, int i, int len) {
    int k;
    if (i < 1 || i > S->length || i + len - 1 > S->length) {
        //非法情况的处理
        printf("不能删除\n");
    } else {
        for (k = i + len - 1; k < S->length; k++) {
            //中从下标为 i+len-1 开始的元素前移
            S->str[k - len] = S->str[k];
        }
        S->length = S->length - len;
        S->str[S->length] = '\0';
        //置字符串S新的结束符
        printf("删除子串后的串是： ");
        printf("%s\n", S->str);
    }
}

//取子串算法
SeqString *subString(SeqString S, int p, int l) {
    int i;
    SeqString *r;
    //处理字符数组为空的情况
    if (S.length <= 0) {
        printf("无子串可取!");
        return (NULL);
    } else {
        r = (SeqString *) malloc(sizeof(SeqString));
        for (i = p; i < p + l; i++) r->str[i - p] = S.str[i];
        r->length = l;
        r->str[r->length] = '\0';
    }
    printf("取出的子串是:");
    printf("%s\n", r->str);
    return (r);
}

void main() {
    SeqString x, y;
    char a[MAXSIZE], b[MAXSIZE];
    int alen, blen, p, l, i;
    char j, sele;
    SeqString *r;
    p = l;
    sele = 'O';
    j = 'l';
    printf("字符串基本操作练习\n");
    printf("字符串基本操作练习\n");
    printf("* * * * * * * * * * *\n");
    printf(" 1--------求串长\n");
    printf(" 2--------串插入\n");
    printf(" 3--------串连接\n");
    printf(" 4--------子串删除\n");
    printf(" 5--------取子串\n");
    printf(" 0--------退出\n");
    printf("* * * * * * * * * * *\n");
    printf("“请选择操作类型:");
    do {
        scanf("%c", &sele);
        switch (sele) {
            case '1':
                printf("请输入串:");
                scanf("%s", a);
                strcpy(x.str, a);
                strLength(x);
                printf("请继续选择:");
                break;
            case '2':
                printf("插入串是：");
                scanf("%s", b);
                blen = strlen(b);
                printf("串长是： %d\n", blen);
                printf("插入到串:");
                scanf("%s", a);
                alen = strlen(a);
                printf("串长是： %d\n", alen);
                printf("请确定插入的位置:");
                scanf("%d", &p);
                x.length = alen;
                y.length = blen;
                strcpy(x.str, a);
                strcpy(y.str, b);
                strInsert(x, p, y); //插入串
                printf("请继续选择: ");
                break;
            case '3':
                printf("前串是：");
                scanf("%s", a);
                printf("后串是：");
                scanf("%s", b);
                strcpy(x.str, a);
                strcpy(y.str, b);
                strConConcat(x, y); //连接 x, y
                printf("请继续选择:");
                break;
            case '4':
                printf("请输入源串:");
                scanf("%s", a);
                r = (SeqString *) malloc(sizeof(SeqString));
                strcpy(x.str, a);
                x.length = strlen(a);
                r->length = x.length;
                for (i = 0; i < x.length; i++) {
                    r->str[i] = x.str[i];
                }
                printf("请确定被删子串的起始位置:");
                scanf("%d", &p);
                printf("请确定被删子串的长度:");
                scanf("%d", &l);    
                strDelete(r, p, l);
                printf("请继续选择:");
                break;
            case '5':
                printf("请输入源串:");
                scanf("%s", a);
                strcpy(x.str, a);
                x.length = strlen(a);
                printf("从第几个字符后取：");
                scanf(" %d", &p);
                printf("确定子串的长度：");
                scanf("d%", &l);
                subString(x, p, l);
                printf("请继续选择：");
                break;
            case '0':
                j = 0;
                printf("\t 程序结束＼n");
                break;
        }
    } while (j == '1');
}
```

## 实验4.2 实现三元组顺序表表示的稀疏矩阵的转置运算

### 考核内容

+   稀疏矩阵采用三元组顺序表的存储结构
+   能够实现按列序为主序进行转置
+   输出矩阵转置前后的三元组

### 程序实现

```C
//
// Created by 张子栋 on 2021/10/15.
//

#include "stdio.h"

#define MAX 20
typedef struct {
    //非零元素的行号
    int row;
    //非零元素的列号
    int col;
    //非零元素的值
    int val;
} TriNode;
typedef struct {
    TriNode data[MAX + 1];
    int m, n, t;
} TriTable;

int main() {
    int i, j, cnt = 1;
    int col, row, val;
    TriTable s;
    TriTable t;
    printf("请输入行数,列数和非零元素个数(以逗号隔开):");
    scanf("%d, %d,%d", &s.m, &s.n, &s.t);
    if (s.t == 0) {
        return 0;
    }
    val = s.t;
    for (i = 1; i <= val; i++) {
        printf("第%d个元素(行号i,列号j,值v)：", i);
        scanf("%d,%d,%d", &s.data[i].col, &s.data[i].val);
    }
    row = s.m;
    col = s.n;
    cnt = 1;
    for (i = 1; i <= col; i++) {
        for (j = 1; j <= val; j++)
            if (s.data[j].col == i) {
                t.data[cnt].row = s.data[j].col;
                t.data[cnt].col = s.data[j].row;
                t.data[cnt++].val = s.data[j].val;
            }
    }
    t.m = s.n;
    t.n = s.m;
    t.t = s.t;
    printf("转置后的三元组:\n");
    for (i = 1; i <= t.t; i++) {
        printf("%d,%d,%d\n", t.data[i].row, t.data[i].col, t.data[i].val);
    }
}
```

### 程序分析

>   这一程序的功能是实现三元组表示的稀疏矩阵的转置，采用三元组顺序表存储。运行时先输入稀疏矩阵的行数和列数及非零元素的个数，再按提示输入各非零元素的 行号、列号、值。
