# 1 创建并输出二叉树

+   采用链式储存表示
+   以广义表形式输入二叉树，使用栈保存二叉树的根结点
+   输出方式采用凹入表示法

```c
#include "stdio.h"
#include "stdlib.h"

#define MAXSIZE 255
typedef struct BiNode {
    char data;
    struct BiNode *lchild;
    struct BiNode *rchild;
} BiTree;

BiTree *CreatBiTreepre(char *str) {
    BiTree *bt, *stack[MAXSIZE], *p = NULL;
    int top = -1, k, j = 0;
    char ch;
    bt = NULL;
    ch = str[j];
    while (ch != '\\0') {
        switch (ch) {
            case '(': {
                top++;
                stack[top] = p;
                k = 1;
                break;
            }
            case ')': {
                top--;
                break;
            }
            case ',': {
                k = 2;
                break;
            }
            default: {
                p = (BiTree *) malloc(sizeof(BiTree));
                p->data = ch;
                p->lchild = p->rchild = NULL;
                if (bt == NULL)
                    bt = p;
                else {
                    switch (k) {
                        case 1:
                            stack[top]->lchild = p;
                            break;
                        case 2:
                            stack[top]->rchild = p;
                            break;
                    }
                }
            }
        }
        j++;
        ch = str[j];
    }
    return bt;
}
```

# 3 线索二叉树及遍历

+   采用二叉链表作为存储结构
+   实现对二叉树的线索化
+   中序遍历线索二叉树
+   在理解递归过程的基础上，尝试采用非递归算法实现二叉树遍历

```c
#include <stdio.h>
#include "malloc.h"

#define ERROR 0
#define OK 1
typedef char TElemType;
typedef enum PointerTag {
    Link, Thread
}; //Link == 0：指针， Thread == 1：线索
typedef struct BiThrNode {
    TElemType data;
    struct BiThrNode *lchild, *rchild;
    enum PointerTag LTag, RTag;
} BiThrNode, *BiThrTree;
BiThrTree pre;

void InOrderTravers_Thr(BiThrTree T) {
    //中序遍历二叉线索树 T 的非递归算法
    BiThrNode *p;
    p = T->lchild;
    while (p != T) {
        while (p->LTag == Link) {
            p = p->lchild;
        }
        printf("%c", p->data);
        while (p->RTag == Thread && p->rchild != T) {
            p = p->rchild;
            printf("%c", p->data);
        }
        p = p->rchild;
    }
}

void InThreading(BiThrTree p) {
    if (p) {
        InThreading(p->lchild);
        if (!p->lchild) {
            p->LTag = Thread;
            p->lchild = pre;
        } else {
            p->LTag = Link;
        }
        if (!pre->rchild) {
            pre->RTag = Thread;
            pre->rchild = p;
        } else {
            p->RTag = Link;
        }
        pre = p;
        InThreading(p->rchild);
    }
}

int InOrderThreading(BiThrTree Thrt, BiThrTree T) {
    if (!(Thrt = (BiThrTree) malloc(sizeof(BiThrNode)))) {
        return ERROR;
    }
    Thrt->LTag = Link;
    Thrt->rchild = Thrt;
    if (!T) {
        Thrt->lchild = Thrt;
        Thrt->RTag = Thread;
    } else {
        Thrt->lchild = T;
        pre = Thrt;
        InThreading(T);
        pre->RTag = Thread;
        Thrt->rchild = pre;
    }
    return OK;
}
```
