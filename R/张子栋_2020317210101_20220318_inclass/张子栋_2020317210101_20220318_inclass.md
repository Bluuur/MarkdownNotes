```R
data <- read.table("hg19_gene_table.txt", header = T)
chromoLength <- read.table('hg19.chrom.sizes.txt', header = F)
geneNum <- table(data$chrom)
geneNum <- geneNum[as.vector(chromoLength$V1)]
geneNum / (chromoLength$V2 / 1000000)
```
